1 "use strict";
  2 
  3 
  4 function LoginPresenter(display, model, callback) {
  5   this.display = display;
  6   this.model = model;
  7   this.callback = callback;
  8 
  9   this.display.onLogin(function(fields) {
 10     display.setEnabled(false);
 11     model.doLogin(fields.email, fields.password, function(err, result) {
 12       display.setEnabled(true);
 13 
 14       if (err) {
 15         alert(err.message);
 16         return true;
 17       } else {
 18         callback();
 19       }
 20     });
 21   });
 22 
 23   this.display.onRegister(function(fields) {
 24     display.setEnabled(false);
 25     model.doRegister(fields.email, fields.password, function(err, result) {
 26       display.setEnabled(true);
 27       if (err) {
 28         alert(err.message);
 29         return true;
 30       } else {
 31         callback();
 32       }
 33     });
 34   });
 35 
 36   display.show();
 37 }
